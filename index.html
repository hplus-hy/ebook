<script> 

  // 총 페이지 수 

  const totalPages = 8; 

 

  // 이미지 기본 URL 

  const baseURL = "https://oobiofficial.speedgabia.com/test/pages/"; 

 

  // 이미지 붙이기 

  for (let i = 1; i <= totalPages; i++) { 

    const num = String(i).padStart(2, '0'); 

    $("#flipbook").append(` 

      <div class="page"> 

        <img src="${baseURL}page-${num}.jpg" /> 

      </div> 

    `); 

  } 

 

  // 플립북 실행 

  $("#flipbook").turn({ 

    width: 800, 

    height: 600, 

    autoCenter: true, 

    gradients: true, 

    elevation: 50 

  }); 

</script>
